void sortArray(int arr[], int size) {
    for (int i = 0; i < size - 1; ++i) {
        for (int j = 0; j < size - i - 1; ++j) {
            if (arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }
}

long long calculateProductDivisibleBy5(int arr[], int size) {
    long long product = 1;
    bool found = false;

    for (int i = 0; i < size; ++i) {
        // –ü–µ—Ä–µ–≤—ñ—Ä—è—î–º–æ, —á–∏ –¥—ñ–ª–∏—Ç—å—Å—è –µ–ª–µ–º–µ–Ω—Ç –Ω–∞ 5 –±–µ–∑ –æ—Å—Ç–∞—á—ñ
        if (arr[i] % 5 == 0) {
            product *= arr[i];
            found = true;
        }
    }

    if (!found) {
        return 0;
    }
    return product;
}
